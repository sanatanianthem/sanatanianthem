Complete technical reference for building and deploying the SanataniAnthem spiritual platform — Website, Android App, iOS App, Admin Panel, and Backend API.

1. System Architecture Overview

All platforms share a single Firestore database. When the Admin updates any content, it propagates to the website via real-time listeners and to mobile apps via FCM + Firestore SDK.

Client Layer
🌐 Next.js Website 🤖 Flutter Android 🍎 Flutter iOS ⚙️ React Admin Panel
↕ REST API + WebSocket + Firestore Real-time
API Gateway Layer
Auth Service (Firebase Auth + JWT) Content Service (CMS API) Puja Service Donation Service (Razorpay) Notification Service (FCM) Media Service (S3/Firebase Storage)
↕ Firebase SDK / Node.js REST
Database Layer
Firestore (Primary — real-time) Firebase Storage (Media) Firebase Auth (Users)
↕ Cloud Functions
Infrastructure
Firebase Hosting Cloud Functions (Node.js) FCM Push Notifications Razorpay Payments YouTube Live API AWS S3 (Media Backup)

5. Key Code Snippets

Firestore Real-time CMS Listener (TypeScript)

// hooks/useCMS.ts
import { doc, onSnapshot } from 'firebase/firestore';
import { db } from '@/lib/firebase';
import { useEffect, useState } from 'react';

export function useCMS() {
  const [config, setConfig] = useState(null);
  const [sections, setSections] = useState({});
  const [navLinks, setNavLinks] = useState([]);

  useEffect(() => {
    // Real-time listener — website auto-updates when admin changes anything
    const unsubSections = onSnapshot(doc(db, 'cms', 'sections'), (snap) => {
      if (snap.exists()) setSections(snap.data());
    });

    const unsubNav = onSnapshot(doc(db, 'cms', 'navLinks'), (snap) => {
      if (snap.exists()) {
        const sorted = [...snap.data().links].sort((a, b) => a.order - b.order);
        setNavLinks(sorted.filter(n => n.visible));
      }
    });

    return () => { unsubSections(); unsubNav(); };
  }, []);

  return { sections, navLinks };
}

Dynamic Section Builder (React)

// components/SectionBuilder.tsx
import { useCMS } from '@/hooks/useCMS';
import DarshanSection from './sections/DarshanSection';
import PujaSection from './sections/PujaSection';
// ... other section imports

const SECTION_MAP = {
  darshan:   DarshanSection,
  puja:      PujaSection,
  services:  ServicesSection,
  astrology: AstrologySection,
  donation:  DonationSection,
  festivals: FestivalsSection,
  knowledge: KnowledgeSection,
  app:       AppSection,
};

export default function SectionBuilder() {
  const { sections } = useCMS();

  const sorted = Object.entries(sections)
    .filter(([_, s]) => s.visible)
    .sort(([_, a], [__, b]) => a.order - b.order);

  return (
    <>
      {sorted.map(([key]) => {
        const Component = SECTION_MAP[key];
        return Component ? <Component key={key} /> : null;
      })}
    </>
  );
}

Flutter CMS Provider (Riverpod)

// lib/providers/cms_provider.dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:cloud_firestore/cloud_firestore.dart';

// Real-time CMS config stream
final cmsConfigProvider = StreamProvider<Map<String, dynamic>>((ref) {
  return FirebaseFirestore.instance
    .collection('cms')
    .doc('sections')
    .snapshots()
    .map((snap) => snap.data() ?? {});
});

// Dynamic home screen sections
final homeSectionsProvider = Provider<List<String>>((ref) {
  final config = ref.watch(cmsConfigProvider);
  return config.when(
    data: (data) {
      return data.entries
        .where((e) => e.value['visible'] == true)
        .toList()
        ..sort((a, b) => (a.value['order'] as int).compareTo(b.value['order']))
        ..map((e) => e.key).toList();
    },
    loading: () => [],
    error: (_, __) => [],
  );
});

// lib/screens/home_screen.dart
import 'package:flutter/material.dart';

Admin Section Toggle + Reorder (Firestore write)

// admin: Update section visibility
async function toggleSection(sectionKey: string, visible: boolean) {
  await updateDoc(doc(db, 'cms', 'sections'), {
    [`${sectionKey}.visible`]: visible,
    updatedAt: serverTimestamp()
  });
  // Website + Apps update automatically via real-time listeners!
}

// admin: Save new section order after drag-drop
async function saveOrder(orderedKeys: string[]) {
  const updates: Record<string, number> = {};
  orderedKeys.forEach((key, index) => {
    updates[`${key}.order`] = index + 1;
  });
  await updateDoc(doc(db, 'cms', 'sections'), updates);
}

// admin: Add new section
async function addSection(section: NewSection) {
  const sectionsDoc = await getDoc(doc(db, 'cms', 'sections'));
  const currentData = sectionsDoc.data() || {};
  const maxOrder = Math.max(...Object.values(currentData).map((s: any) => s.order || 0), 0);

  await updateDoc(doc(db, 'cms', 'sections'), {
    [section.key]: {
      name: section.name,
      icon: section.icon,
      visible: true,
      order: maxOrder + 1,
      createdAt: serverTimestamp()
    }
  });
  // Appears on Website, Android, iOS immediately!
}

FCM Push Notification (Cloud Function)

// functions/src/routes/notifications.ts
import * as admin from 'firebase-admin';
import express from 'express';

const router = express.Router();

router.post('/send', adminOnly, async (req, res) => {
  const { title, body, target, imageUrl, link } = req.body;

  const message: admin.messaging.MulticastMessage = {
    notification: { title, body, imageUrl },
    data: { link: link || '/', click_action: 'FLUTTER_NOTIFICATION_CLICK' },
    android: { notification: { channelId: 'sanatani_main', color: '#FF6B00' } },
    apns: { payload: { aps: { badge: 1, sound: 'default' } } },
    tokens: await getTokensByTarget(target) // 'all' | 'android' | 'ios' | 'web'
  };

  const response = await admin.messaging().sendEachForMulticast(message);

  // Log to Firestore
  await admin.firestore().collection('notification_logs').add({
    title, body, target, sentAt: admin.firestore.FieldValue.serverTimestamp(),
    successCount: response.successCount, failureCount: response.failureCount
  });

  res.json({ success: true, sent: response.successCount });
});

2
Configure Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // CMS — only admins write
    match /cms/{doc} {
      allow read: if true;
      allow write: if request.auth != null && 
                      get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    // Temples — public read
    match /temples/{id} {
      allow read: if true;
      allow write: if isAdmin();
    }
    // Users — read own, admin reads all
    match /users/{uid} {
      allow read, write: if request.auth.uid == uid;
      allow read: if isAdmin();
    }
    // Bookings — user owns their bookings
    match /bookings/{id} {
      allow read, write: if request.auth != null && 
                            request.auth.uid == resource.data.userId;
      allow read, write: if isAdmin();
    }
    function isAdmin() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
  }
}
